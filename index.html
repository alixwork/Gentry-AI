async function runAgent() { const productUrl = document.getElementById('productLink').value; const chat = document.getElementById('chat-area'); const btn = document.getElementById('mainBtn'); if (!productUrl) { chat.innerHTML = "Gentry: Te rog să introduci un link."; return; } btn.disabled = true; btn.innerHTML = ' ANALIZĂ...'; try { const response = await fetch('https://gentry-ai.onrender.com/analyze', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ url: productUrl }) }); const data = await response.json(); if (data.status === "Success") { saveToWishlist(data.product_name, data.price, productUrl); chat.innerHTML = `Gentry: Produs găsit! Preț: ${data.price} lei.`; } else { // PLANUL B: Dacă serverul e blocat, cerem datele manual ca să nu blocăm utilizatorul let manualName = prompt("eMAG este ocupat. Introdu numele produsului:"); let manualPrice = prompt("Introdu prețul văzut de tine (cifre):"); if(manualName && manualPrice) { saveToWishlist(manualName, manualPrice, productUrl); chat.innerHTML = "Gentry: Am salvat produsul manual în portofoliu."; } } } catch (error) { // Dacă serverul e picat de tot let manualName = prompt("Conexiune lentă. Introdu numele produsului:"); let manualPrice = prompt("Introdu prețul:"); if(manualName && manualPrice) { saveToWishlist(manualName, manualPrice, productUrl); chat.innerHTML = "Gentry: Produs salvat local."; } } finally { btn.disabled = false; btn.innerText = "SCANEAZĂ PRODUSUL"; document.getElementById('productLink').value = ""; displayWishlist(); } }
